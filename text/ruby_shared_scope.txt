h3. Ruby Shared Scope:
<pre>
#!/usr/env/ruby
def define_methods
  shared = 0
  Kernel.send :define_method, :counter do
    shared
  end
  Kernel.send :define_method, :inc do |x|
    shared += x
  end
end

define_methods
puts counter  #=> 0
puts inc(4)  #=> 4
puts counter  #=> 4
